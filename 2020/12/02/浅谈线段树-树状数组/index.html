线段树简介

线段树，一种二叉树（我并不明白为什么一些人认为她是二叉搜索树）就是把一个区间看成二叉树的一个节点，把一个区间不停从尽量靠中间的位置分段，且把两个（也可能是一个）子区间作为此区间的左右两个二子的一种数据结构。

就是这个样子：

她可以做到在的时间复杂度内做到处理一次区间询问和一次区间修改（前提是保证可行性和正确性）。树套树什么的可能要套一些log

but建树需要O(nlogn)而且只能离线做（一些奇技淫巧可以在线线段树）。

例一

单点加和询问区间和。

解决这个问题我们的思路是每个节点维护这个区间的和，显然单点修改+回溯过程中维护区间和的复杂度是O(logn)。因为我们每次把一层的每一个区间分成两个，最多分logn次也就是线段树的深度为logn。

然后就是区间查询该怎么处理？

我们从线段树根节点开始，查看当前节点是否被查询区间完全覆盖，如果是那么把当前节点统计的和加入答案并结束递归，否则就遍历两个儿子。可以发现这样每一层最多遍历到4个节点，复杂度O(logn)。

至此，这个问题得到完美解决。

总的来说，我们用到了线段树两个非常好的性质。

具体代码如下：

#include<iostream>
#include<string>
#include<cmath>
#include<cstdio>
#include<algorithm>
using namespace std;
#define N 501000
#define ls now<<1
#define rs now<<1|1 
#define mid ((L+R)>>1) 
int n,m,sum[N*20];
int read(){
	int sum=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9'){if(ch=='-')f=-1;ch=getchar();}
	while(ch>='0'&&ch<='9'){sum=sum*10+ch-'0';ch=getchar();}
	return sum*f;
}
void update(int now){sum[now]=sum[ls]+sum[rs];}
void add(int L,int R,int x,int w,int now){
	if(L==R){sum[now]+=w;return;}
	if(x>mid)add(mid+1,R,x,w,rs);
	else if(x<=mid)add(L,mid,x,w,ls);
	update(now); 
}
int check(int L,int R,int l,int r,int now){
	if(L==l&&R==r){return sum[now];}
	if(l>mid)return check(mid+1,R,l,r,rs);
	else if(r<=mid)return check(L,mid,l,r,ls);
	else return check(L,mid,l,mid,ls)+check(mid+1,R,mid+1,r,rs);
}
int main(){
	n=read(),m=read();
	for(int i=1;i<=n;i++)add(1,n,i,read(),1);
	while(m--){
		int type=read();
		if(type==1){
			int x=read(),w=read();
			add(1,n,x,w,1);
		}
		else {
			int l=read(),r=read();
			printf("%d\n",check(1,n,l,r,1)); 
		}
	}
	return 0;
}

例二

区间加和询问区间和。

区间加该如何处理？利用询问区间和的方法遍历组成这个区间的节点，然后修改其区间和的方法是不行的.
比如：

红色是区间加最后修改的节点，而蓝色的是询问区间对应的两个节点。
发现蓝色的4节点并没有被修改，因而这种方法无法实现。

这个时候就要引入一个新的东西——lazy标记。

具体是这样的：每一个节点都对应一个lazy值，代表此节点的后代都需要有一个+lazy的操作,我们在执行修改操作时，在不停的递归中给经过的节点修改sum，在最后拼成询问区间的若干个节点的lazy加上w。在查询操作中，我们将递归过程中每个经过的节点的lazy下移。这样就可以解决上述问题。而且不增加复杂度的数量级。

代码实现如下：

#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
#define N 100100
#define ls now<<1
#define rs now<<1|1
#define mid ((L+R)>>1)
#define int long long
int n,m,sum[N*20],lazy[N*20];
int read(){
	int sum=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9'){if(ch=='-')f=-1;ch=getchar();}
	while(ch>='0'&&ch<='9'){sum=sum*10+ch-'0';ch=getchar();}
	return sum*f;
}
void pushdown(int L,int R,int now){
	if(lazy[now]==0)return;
	sum[ls]+=(mid-L+1)*lazy[now];
	sum[rs]+=(R-mid)*lazy[now];
	lazy[ls]+=lazy[now];
	lazy[rs]+=lazy[now];
	lazy[now]=0;
}
void update(int now){sum[now]=sum[ls]+sum[rs];}
void add(int L,int R,int l,int r,int w,int now){
	sum[now]+=(r-l+1)*w;
	if(L==l&&R==r){
		lazy[now]+=w;
		return;
	}
	if(l>mid)add(mid+1,R,l,r,w,rs);
	else if(r<=mid)add(L,mid,l,r,w,ls);
	else add(L,mid,l,mid,w,ls),add(mid+1,R,mid+1,r,w,rs);
}
int check(int L,int R,int l,int r,int now){
	pushdown(L,R,now);
	if(L==l&&R==r)return sum[now];
	if(l>mid)return check(mid+1,R,l,r,rs);
	else if(r<=mid)return check(L,mid,l,r,ls);
	else return check(L,mid,l,mid,ls)+check(mid+1,R,mid+1,r,rs);
}
signed main(){
	n=read(),m=read();
	for(int i=1;i<=n;i++)add(1,n,i,i,read(),1);
	while(m--){
		int type=read();
		if(type==1){
			int l=read(),r=read(),w=read();
			add(1,n,l,r,w,1);
		}
		else{
			int l=read(),r=read();
			printf("%lld\n",check(1,n,l,r,1));
		}
	}
	return 0;
}

这算是线段树比较经典的两个应用了。
灵活利用线段树的一些很好的性质和lazy标记的应用。她就可以做到更多的东西。不一定维护数列，也可以维护矩阵和一些你根本想不到的东西。

树状数组简介

树状数组，同样是树形数据结构，十分简介，实现十分巧妙，完美地利用了补码的性质和位运算。

话不多说，先上图。

可以看到，a[i]数组即为原数组，c[i]数组就是我们要创建和维护的树状数组数组（姑且这么叫）。c[i]数组记录其儿子的和，利用一些奇技淫巧可以O(logn)地维护它。

lowbit

这是树状数组的灵魂。lowbit(i)会返回i对应二进制最末位的1。比如lowbit(6)=2,6的二进制表示为110，2的二进制表示是10，对应6的二进制110最末位的1。

lowbit的计算方法是lowbit(i)=i&-i

这是利用了补码的性质：负数补码为反码+1。设一个负数原码最末x位为0则倒数x+1末位为1，反码最后x位就为1且倒数x+1位为0，补码把反码加1后与原码取按位和，正好可以把最后x+1位求出，因为原码和补码只有倒数第x+1位全是1。

那么lowbit有什么用呢？

可以在上图中发现，对于任何一个数i，c[i+lowbit[i]]就是c[i]的父节点。所以维护只需要O(logn)。

例如求前7个数的和，直接求c[7]+c[6]+c[4]即可。而7-lowbit(7)=6,6-lowbit(6)=4!求前缀只需要O(logn)。

为什么树状数组有这么好的性质？

c[i]数组其实维护的是a数组第i个位置前lowbit(i)个数的和。设i的二进制最后几位为101100，则c[i]就是a[101000]到a[101100]的和，因为在101000和101100之间的数在+若干次lowbit后一定会变成101100，而除此之外的任何数都无法到达101100(这点在试几个数后就可以意会，就不证明了)。

就是因为这样我们要求前i个数的和的话，只需要一直令i=i-lowbit(i),然后把c[i]加起来就行了。

应用

单点加，区间求和是典型应用。

#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
#define N 501000
int n,m,c[N];
int lowbit(int x){return x&-x;}
void add(int x,int w){
	for(int i=x;i<=n;i+=lowbit(i))
		c[i]+=w;
}
int getsum(int x){
	if(x<1)return 0;
	return c[x]+getsum(x-lowbit(x));
}
int read(){
	int sum=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9'){if(ch=='-')f=-1;ch=getchar();} 
	while(ch>='0'&&ch<='9'){sum=sum*10+ch-'0';ch=getchar();}
	return sum*f;
}
int main(){
	n=read(),m=read();
	for(int i=1;i<=n;i++)add(i,read());
	while(m--){
		int type=read();
		if(type==1){
			int x=read(),w=read();
			add(x,w);
		}
		else {
			int l=read(),r=read();
			printf("%d\n",getsum(r)-getsum(l-1));
		}
	}
	return 0;
}

		c[i]+=w;
}

int read(){
	int sum=0,f=1;
	char ch=getchar();

int main(){
	n=read(),m=read();

		int w=read();

		}

其实还可以做到区间加区间查询，但是有些复杂，失去了树状数组简洁美丽的特点，可以用线段树值班。

虽然线段树的功能包含了树状数组，但是树状数组的常数小，空间小，在树套树中的时空表现优于线段树。
